## Example

```rust
use docx_rs::*;

pub fn hello() -> Result<(), DocxError> {
  let path = std::path::Path::new("./tests/output/hello.docx");
  let file = std::fs::File::create(&path).unwrap();
 Docx::new()
    .add_paragraph(Paragraph::new().add_run(Run::new().add_text("Hello")))
    .build()
    .pack(file)?;
  Ok(())
}
```
